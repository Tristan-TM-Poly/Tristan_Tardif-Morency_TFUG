      - name: Set up Python
        if: steps.has_dvc.outputs.present == 'true'
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Install DVC
        if: steps.has_dvc.outputs.present == 'true'
        run: |
          pip install "dvc[s3,gdrive,azure,gs,oss,webdav]"

      - name: Configure DVC remote (optional)
        if: steps.has_dvc.outputs.present == 'true'
        env:
          DVC_REMOTE: ${{ secrets.DVC_REMOTE }}
        run: |
          if [ -n "${DVC_REMOTE}" ]; then
            dvc remote add -f ci-remote "${DVC_REMOTE}" || true
            dvc remote default ci-remote || true
          else
            echo "No DVC_REMOTE secret — operating offline."
          fi

      - name: DVC status / pull (best-effort)
        if: steps.has_dvc.outputs.present == 'true'
        run: |
          dvc status || true
          dvc pull || true

      - name: Validate dataspec (if present)
        if: steps.has_dvc.outputs.present == 'true'
        run: |
          if [ -f dataspec.yaml ]; then python -c "import yaml,sys; yaml.safe_load(open('dataspec.yaml')); print('dataspec ✅')" || exit 1; else echo "No dataspec.yaml — skipping"; fi

4) Bump de version simple (facultatif)
======================================

`tools/bump_version.sh`
-----------------------

```bash
#!/usr/bin/env bash
set -euo pipefail
PART="${1:-patch}"  # major|minor|patch
FILE="pyproject.toml"

current=$(grep -E '^version\s*=\s*"' "$FILE" | sed -E 's/.*"([^"]+)".*/\1/')
IFS='.' read -r MA MI PA <<<"$current"

case "$PART" in
  major) MA=$((MA+1)); MI=0; PA=0;;
  minor) MI=$((MI+1)); PA=0;;
  patch) PA=$((PA+1));;
  *) echo "Use: bump_version.sh [major|minor|patch]"; exit 1;;
esac

new="${MA}.${MI}.${PA}"
sed -i.bak -E "s/^version\s*=\s*\"[^\"]+\"/version = \"$new\"/" "$FILE" && rm -f "$FILE.bak"
echo "Bumped: $current → $new"
```

#!/usr/bin/env bash
set -euo pipefail
PART="${1:-patch}"  # major|minor|patch
FILE="pyproject.toml"
current=$(grep -E '^version\s*=\s*"' "$FILE" | sed -E 's/.*"([^"]+)".*/\1/')
IFS='.' read -r MA MI PA <<<"$current"
case "$PART" in
  major) MA=$((MA+1)); MI=0; PA=0;;
  minor) MI=$((MI+1)); PA=0;;
  patch) PA=$((PA+1));;
  *) echo "Use: bump_version.sh [major|minor|patch]"; exit 1;;
esac
new="${MA}.${MI}.${PA}"
sed -i.bak -E "s/^version\s*=\s*\"[^\"]+\"/version = \"$new\"/" "$FILE" && rm -f "$FILE.bak"
echo "Bumped: $current → $new"
SH
